import pandas as pd
from pathlib import Path

# Load redacted quarterly dataset
path = Path('data/earnings.csv')
df = pd.read_csv(path)

# Simple growth calc
q_rev = df.groupby('quarter')['revenue_cr'].sum()
growth = q_rev.pct_change().round(3)
print('QoQ growth:', growth.iloc[-1])
